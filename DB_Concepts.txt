BETWEEN- Any particular range
select * from payment where payment_date BETWEEN '25-04-2026' AND 29-04-2026';

NOT BETWEEN- Not in range
select * from payment where payment_date BETWEEN '25-04-2026' AND 29-04-2026';

IN- Use for any column value
select * from payment where amount IN(0.99,1.98,1.99);
select COUNT(*) from payment where amount IN(0.99,1.98,1.99); -- it will give count of payment for these amount
select * from Customer where first_name IN('John', 'Jake');

NOT IN - value not in column
select * from payment where amount IN(0.99,1.98,1.99);

LIKE- case sensitive
ALIKE- not case sensitive
select * from Customer where first_name LIKE 'J%' AND last_name LIKE 'S%';  o/p- name starts with J
select * from Customer where first_name LIkE '%er%'; o/p- er is somewhere in the name eg Jennifer
select * from Customer where first_name LIkE '_her%'; o/p it allows only one character before her eg sherry
